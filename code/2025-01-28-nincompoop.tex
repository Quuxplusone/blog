\newcommand\PuzzleBox[2][]{%
  \def\Puzzletmp{#2}%
  \if\Puzzletmp.
    \PuzzleX=0\relax \advance\PuzzleY-1
  \else
    \ifx\empty\Puzzletmp
    \else\if\Puzzletmp *
      \put(\PuzzleX,\PuzzleY){\framebox(1,1){\PuzzleBlackBox}}
    \else
      \put(\PuzzleX,\PuzzleY){\framebox(1,1){}}
    \fi\fi
    \def\Puzzletmp{#1}%
    \ifx\empty\Puzzletmp
    \else
      \put(\PuzzleX,\PuzzleY){\makebox(1,.9)[tl]{\hspace{.08\PuzzleUnitlength}\PuzzleNumberFont #1}}
    \fi
    \def\Puzzletmp{#2}%
    \if\Puzzletmp X
    \else
      \put(\PuzzleX,\PuzzleY){\makebox(1,.7){\hspace{.08\PuzzleUnitlength}\PuzzleSolutionFont #2}}
    \fi
    \advance\PuzzleX 1
  \fi
}
